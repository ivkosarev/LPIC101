2) Смонтировать в них несколько пустых корневых каталогов.
mount /dev/sdb1 /

3) Перенести каталог /home на отдельный раздел.
sudo -s # нужен рут
mkfs.ext4 /dev/sdb1 # форматнем раздел
mkdir /mnt/home 
mount /dev/sdb1 /mnt/home  # примонтировали каталог
cp -a /home/* /mnt/home # скопировали все из /home в /mnt
rm -rf /home/* # удалили старую папку /home
echo "/dev/sdb1       /home           ext4    defaults           0       1" >> /etc/fstab # допишем в fstab новое положение /home
mount -a # примонтировали все файловые системы что были указаны в fstab
df -h | grep /home # проверяем, на выходе имеем:
/dev/sdb1        1.9G          12M  1.7G            1% /home

4) Создать еще один раздел подкачки.

 fallocate -l 1G /swapfile1 # создали второй файл свопа с размером 1гиг
 chmod 600 /swapfile1 # ограничили права всем, кроме рута, руту дали только чтение
 mkswap /swapfile1 # отметили новый файл как пространство подкачки
 swapon /swapfile1 # автивировали своп
 swapon --show # проверили, итого имеем:
 NAME       TYPE   SIZE USED PRIO
/swapfile  file 947.2M   0B   -2
/swapfile1 file  1024M   0B   -3   https://github.com/Nefelim4ag/systemd-swap

vgcreate vg2 /dev/sdb2 /dev/sdc2

lvcreate -n lv2 -L 1G vg2 # создали логический lv2 раздел с указанием размера и виртуальной группы
